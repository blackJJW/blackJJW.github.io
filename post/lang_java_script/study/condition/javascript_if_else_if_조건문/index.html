    <script>
	const date = new Date() // 현재 날짜와 시간을 갖는 객체 생성
  const hour = date.getHours() // 현재시간을 0 ~ 23 사이의 값으로 출력하는 메소드

  if (hour < 11) {
	  alert('아침 식사 시간')

	} else {
	  alert('저녁 식사 시간')
  }
</script>
